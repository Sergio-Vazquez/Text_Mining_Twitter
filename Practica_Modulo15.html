PROCESO DE NORMALIZACION DE DATOS

# Quitar espacios en blanco extra:
corpus.Podemos <- tm_map(corpus.Podemos, stripWhitespace, lazy=TRUE)

# Pasamos a minusculas:
corpus.Podemos <- tm_map(corpus.Podemos, content_transformer(tolower), lazy=TRUE)

# Eliminamos numeros:
corpus.Podemos <- tm_map(corpus.Podemos, removeNumbers, lazy=TRUE)

# Eliminamos signos de puntuacion:
corpus.Podemos <- tm_map(corpus.Podemos, removePunctuation, lazy=TRUE)

# Eliminamos "stopwords":
corpus.Podemos <- tm_map(corpus.Podemos, removeWords, stopwords("es"), lazy=TRUE)

# Lo pasamos a "PlainTextDocument" para que no de problemas al crear la matriz de terminos:
corpus.Podemos <- tm_map(corpus.Podemos, PlainTextDocument)

CREACION DE LA MATRIZ DE TERMINOS

# terminos en columnas y documentos en filas
dtm <- DocumentTermMatrix(corpus.Podemos,
                         control = list(removePunctuation = TRUE,
                                        stopwords = stopwords("spanish"),

dtm.matriz <- as.matrix(dtm) #  matriz de 2000 filas X 2780 columnas

USUARIOS MAS ACTIVOS

# Ver los usuarios mas activos en esos dias:
countUser=table(df.tweetsPodemos$screenName)

# Ordenamos los usuarios de más activo a menos:
TopUser = sort(countUser, decreasing=T)
# Obtenemos una primera aproximacion:
barplot(TopUser)

# Creo una paleta de colores:
degradado = function (color1, color2, degradados)
{
  library(grDevices)
  palete = colorRampPalette(c(color1, color2))
  palete (degradados)
}

paleta = degradado ("#DD0000", "#FFFF00", 10)

par(mar=c(10,5,2,2), bg="white") ## Establezco los márgenes para ajustar la imagen

barplot(TopUser[1:10],las=2,cex.names =1, col=paleta) ## Dibujo un plot de los 10 usuarios TOP
